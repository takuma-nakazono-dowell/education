継承とポリモーフィズム

継承の基本

// 基底クラス（親） class Animal { public string Name { get; set; } public Animal(string name) { Name = name; } public virtual void Speak() { Console.WriteLine($"{Name}が鳴いた"); } public void Eat() { Console.WriteLine($"{Name}が食べた"); } } // 派生クラス（子） class Dog : Animal { public Dog(string name) : base(name) { } // overrideで親のメソッドを上書き public override void Speak() { Console.WriteLine($"{Name}：ワン！"); } } class Cat : Animal { public Cat(string name) : base(name) { } public override void Speak() { Console.WriteLine($"{Name}：ニャー！"); } } // 使い方 var dog = new Dog("ポチ"); var cat = new Cat("タマ"); dog.Speak(); // ポチ：ワン！ cat.Speak(); // タマ：ニャー！ dog.Eat(); // ポチが食べた（継承したメソッド）
ポイント: virtual をつけたメソッドだけを override 可能。基底クラスのコンストラクタは base() で呼ぶ。

ポリモーフィズム

// 親クラス型の変数で子クラスを扱える Animal[] animals = { new Dog("ポチ"), new Cat("タマ"), new Dog("ハチ") }; foreach (Animal animal in animals) { animal.Speak(); // それぞれの実装が呼ばれる } // ポチ：ワン！ // タマ：ニャー！ // ハチ：ワン！

sealed（継承禁止）

sealed class FinalClass { } // class SubClass : FinalClass { } // エラー: sealedクラスは継承不可

base キーワード

class ElectricCar : Car { public int BatteryLevel { get; set; } public ElectricCar(string model, int battery) : base(model) { BatteryLevel = battery; } public override void Describe() { base.Describe(); // 親のメソッドも呼ぶ Console.WriteLine($"バッテリー残量: {BatteryLevel}%"); } }

is / as 演算子（型チェック）

Animal animal = new Dog("ポチ"); // is: 型チェックしながらキャスト（C# 7以降のパターンマッチング） if (animal is Dog dog) { Console.WriteLine($"{dog.Name}は犬です"); } // as: キャスト失敗時にnullを返す Dog? maybeDog = animal as Dog; if (maybeDog != null) { /* 犬として扱う */ }

練習問題

問題 1 easy

以下のコードを読んで、出力結果を答えてください。

class Shape { public virtual string GetName() => "図形"; } class Circle : Shape { public override string GetName() => "円"; } Shape s = new Circle(); Console.WriteLine(s.GetName());
答えを見る
円 // 解説: Shape型の変数でも、実際のオブジェクトはCircleなので // overrideされたGetName()が呼ばれる（ポリモーフィズム）

問題 2 easy

空欄を埋めて継承を完成させてください。

class Vehicle { public string Brand { get; } public Vehicle(string brand) { Brand = brand; } public virtual void Move() => Console.WriteLine("移動する"); } class Car : ____ { public Car(string brand) : ____(brand) { } public ____ void Move() => Console.WriteLine("走る"); }
答えを見る
// 空欄の答え（上から順に） // 1番目: Vehicle （継承元クラス名） // 2番目: base （基底クラスのコンストラクタ呼び出し） // 3番目: override （メソッドの上書き） class Car : Vehicle { public Car(string brand) : base(brand) { } public override void Move() => Console.WriteLine("走る"); }

問題 3 medium

Shape 基底クラスと Circle・Rectangle 派生クラスを書いてください。

  • Shape に virtual double Area() を持たせ、それぞれでoverrideする
  • 円の面積 = π × r²
  • 長方形の面積 = 幅 × 高さ
答えを見る
class Shape { public virtual double Area() => 0; } class Circle : Shape { public double Radius { get; } public Circle(double radius) { Radius = radius; } public override double Area() => Math.PI * Radius * Radius; } class Rectangle : Shape { public double Width { get; } public double Height { get; } public Rectangle(double width, double height) { Width = width; Height = height; } public override double Area() => Width * Height; } // 使用例 var circle = new Circle(5); var rect = new Rectangle(4, 6); Console.WriteLine($"円の面積: {circle.Area():F2}"); // 78.54 Console.WriteLine($"長方形の面積: {rect.Area():F2}"); // 24.00

問題 4 medium

問題3で作成したクラスを使い、ポリモーフィズムを活用してください。

  • Shape[] に複数の図形を入れる
  • forループで全図形の面積を合計するコードを書く
答えを見る
// 親クラス型の配列に、異なる子クラスのインスタンスを入れる Shape[] shapes = { new Circle(3), new Rectangle(4, 5), new Circle(2), new Rectangle(6, 2) }; double total = 0; foreach (Shape shape in shapes) { double area = shape.Area(); total += area; Console.WriteLine($"面積: {area:F2}"); } Console.WriteLine($"合計面積: {total:F2}"); // 出力例: // 面積: 28.27 // 面積: 20.00 // 面積: 12.57 // 面積: 12.00 // 合計面積: 72.84

問題 5 hard

従業員管理システムを設計してください。

  • Employee（基底）・Manager・Developer の3クラスを作る
  • 各クラスで GetRole() をoverrideして返す文字列を変える
  • List<Employee> に混在させてforeachで全員の役職を出力する
答えを見る
class Employee { public string Name { get; } public Employee(string name) { Name = name; } public virtual string GetRole() => "一般社員"; } class Manager : Employee { public Manager(string name) : base(name) { } public override string GetRole() => "マネージャー"; } class Developer : Employee { public Developer(string name) : base(name) { } public override string GetRole() => "開発者"; } // ポリモーフィズムの活用 var employees = new List<Employee> { new Employee("田中"), new Manager("山田"), new Developer("佐藤"), new Developer("鈴木"), new Manager("高橋") }; foreach (Employee emp in employees) { Console.WriteLine($"{emp.Name}: {emp.GetRole()}"); } // 出力: // 田中: 一般社員 // 山田: マネージャー // 佐藤: 開発者 // 鈴木: 開発者 // 高橋: マネージャー
